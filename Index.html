                            </tr>
                          </thead>
                          <tbody className="divide-y divide-gray-800/60">
                            {players.map(p => {
                              const computedPayment = p.type === 'monthly' ? p.checks * 100000 : p.checks * 25000;
                              return (
                                <tr key={p.id}>
                                  <td className="py-4 px-6 font-bold text-white text-sm">{p.name}</td>
                                  <td className="py-4 px-6">{p.type === 'monthly' ? 'Bulanan' : 'Per Play'}</td>
                                  <td className="py-4 px-6">{"✅".repeat(p.checks) || "Belum Bayar"}</td>
                                  <td className="py-4 px-6 font-black text-emerald-400">Rp {computedPayment.toLocaleString('id-ID')}</td>
                                  <td className="py-4 px-6 flex justify-center gap-2">
                                    <button onClick={() => incrementCheck(p.id)} className="w-8 h-8 rounded bg-emerald-950 text-emerald-400 font-bold">+✅</button>
                                    <button onClick={() => decrementCheck(p.id)} className="w-8 h-8 rounded bg-red-950 text-red-400 font-bold">-✅</button>
                                    <button onClick={() => removePlayer(p.id, p.name)} className="w-8 h-8 rounded bg-gray-950 text-gray-400">🗑️</button>
                                  </td>
                                </tr>
                              );
                            })}
                          </tbody>
                        </table>
                      </div>
                    </div>
                  </div>
                )}

                {activeTab === 'matchmaker' && (
                  <div className="space-y-6">
                    <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
                      <div className="bg-gray-900 border border-gray-800 p-6 rounded-3xl shadow-xl space-y-4">
                        <h3 className="text-lg font-bold text-white">Pilih Tim Ganda</h3>
                        <select value={dPlayer1} onChange={(e) => setDPlayer1(e.target.value)} className="w-full bg-gray-950 border border-gray-800 rounded-xl p-2.5 text-sm text-white"><option value="">Pemain 1</option>{players.map(p => <option key={p.id} value={p.id}>{p.name}</option>)}</select>
                        <select value={dPlayer2} onChange={(e) => setDPlayer2(e.target.value)} className="w-full bg-gray-950 border border-gray-800 rounded-xl p-2.5 text-sm text-white"><option value="">Pemain 2</option>{players.map(p => <option key={p.id} value={p.id}>{p.name}</option>)}</select>
                        <select value={dPlayer3} onChange={(e) => setDPlayer3(e.target.value)} className="w-full bg-gray-950 border border-gray-800 rounded-xl p-2.5 text-sm text-white"><option value="">Pemain 3</option>{players.map(p => <option key={p.id} value={p.id}>{p.name}</option>)}</select>
                        <select value={dPlayer4} onChange={(e) => setDPlayer4(e.target.value)} className="w-full bg-gray-950 border border-gray-800 rounded-xl p-2.5 text-sm text-white"><option value="">Pemain 4</option>{players.map(p => <option key={p.id} value={p.id}>{p.name}</option>)}</select>
                      </div>

                      <div className="bg-gray-900 border border-gray-800 p-6 rounded-3xl shadow-xl text-center">
                        <h3 className="text-lg font-bold text-white mb-4">Scoreboard Live</h3>
                        <div className="grid grid-cols-2 gap-4">
                          <button onClick={() => setScoreA(prev => prev + 1)} className="text-4xl font-bold bg-indigo-950/60 p-4 rounded-xl border border-indigo-800">{scoreA}</button>
                          <button onClick={() => setScoreB(prev => prev + 1)} className="text-4xl font-bold bg-red-950/60 p-4 rounded-xl border border-red-800">{scoreB}</button>
                        </div>
                        <button onClick={saveMatchResult} className="w-full mt-4 py-3 bg-emerald-500 text-gray-950 font-bold rounded-xl">Simpan Skor</button>
                      </div>

                      <div className="bg-gray-900 border border-gray-800 p-6 rounded-3xl shadow-xl max-h-[300px] overflow-y-auto">
                        <h3 className="text-lg font-bold text-white mb-2">Win Rate</h3>
                        {players.map((p, i) => (
                          <div key={p.id} className="flex justify-between py-1 border-b border-gray-800 text-xs">
                            <span>{p.name}</span>
                            <span>{p.stats?.played ? Math.round((p.stats.won / p.stats.played) * 100) : 0}% WR</span>
                          </div>
                        ))}
                      </div>
                    </div>
                  </div>
                )}

                {activeTab === 'ledger' && (
                  <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
                    <div className="bg-gray-900 border border-gray-800 p-6 rounded-3xl shadow-xl">
                      <form onSubmit={addTransaction} className="space-y-4">
                        <div className="grid grid-cols-2 gap-2">
                          <button type="button" onClick={() => setTxType('credit')} className={`py-2 px-4 rounded-xl text-xs font-bold border ${txType === 'credit' ? 'border-emerald-500 text-emerald-400' : 'border-gray-800'}`}>Masuk</button>
                          <button type="button" onClick={() => setTxType('debit')} className={`py-2 px-4 rounded-xl text-xs font-bold border ${txType === 'debit' ? 'border-red-500 text-red-400' : 'border-gray-800'}`}>Keluar</button>
                        </div>
                        <input type="number" value={txAmount} onChange={(e) => setTxAmount(e.target.value)} placeholder="Jumlah (Rp)" className="w-full bg-gray-950 border border-gray-800 rounded-xl px-4 py-2.5 text-sm text-white" />
                        <input type="text" value={txDesc} onChange={(e) => setTxDesc(e.target.value)} placeholder="Keterangan" className="w-full bg-gray-950 border border-gray-800 rounded-xl px-4 py-2.5 text-sm text-white" />
                        <input type="file" accept="image/*" onChange={handleReceiptUpload} className="w-full text-xs text-gray-400" />
                        <button type="submit" className="w-full py-3 bg-emerald-500 text-gray-950 rounded-xl font-bold">Simpan Buku Kas</button>
                      </form>
                    </div>

                    <div className="lg:col-span-2 space-y-4 max-h-[400px] overflow-y-auto">
                      {transactions.map(t => (
                        <div key={t.id} className="p-4 bg-gray-900 border border-gray-800 rounded-xl flex justify-between items-center">
                          <div>
                            <span className="font-bold text-white block">{t.description}</span>
                            <span className="text-xs text-gray-400">{t.date}</span>
                          </div>
                          <div className="flex items-center gap-3">
                            <span className={t.type === 'credit' ? 'text-emerald-400' : 'text-red-400'}>Rp {t.amount.toLocaleString('id-ID')}</span>
                            {t.slip && <button onClick={() => setSelectedSlip(t.slip)} className="text-xs text-indigo-400 bg-indigo-950/40 p-1.5 rounded">Slip 📸</button>}
                            <button onClick={() => removeTransaction(t.id)} className="text-red-500 text-xs">✕</button>
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>
                )}

                {activeTab === 'tactical' && (
                  <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
                    <div className="bg-gray-900 border border-gray-800 p-6 rounded-3xl shadow-xl space-y-4">
                      <h3 className="text-lg font-bold text-white">Preset Formasi</h3>
                      <div className="grid grid-cols-3 gap-2">
                        <button onClick={() => setTacticalPlayers([{ id: 'tp1', label: 'T1-A', x: 50, y: 200, color: '#3B82F6' }, { id: 'tp2', label: 'T1-B', x: 100, y: 200, color: '#3B82F6' }, { id: 'tp3', label: 'T2-A', x: 200, y: 200, color: '#EF4444' }, { id: 'tp4', label: 'T2-B', x: 250, y: 200, color: '#EF4444' }])} className="py-1 bg-gray-800 text-xs text-white rounded">Defensif</button>
                        <button onClick={() => setTacticalPlayers([{ id: 'tp1', label: 'T1-A', x: 75, y: 150, color: '#3B82F6' }, { id: 'tp2', label: 'T1-B', x: 75, y: 280, color: '#3B82F6' }, { id: 'tp3', label: 'T2-A', x: 225, y: 150, color: '#EF4444' }, { id: 'tp4', label: 'T2-B', x: 225, y: 280, color: '#EF4444' }])} className="py-1 bg-gray-800 text-xs text-white rounded">Ofensif</button>
                        <button onClick={() => setTacticalPlayers([{ id: 'tp1', label: 'T1-A', x: 75, y: 120, color: '#3B82F6' }, { id: 'tp2', label: 'T1-B', x: 40, y: 280, color: '#3B82F6' }, { id: 'tp3', label: 'T2-A', x: 225, y: 120, color: '#EF4444' }, { id: 'tp4', label: 'T2-B', x: 260, y: 280, color: '#EF4444' }])} className="py-1 bg-gray-800 text-xs text-white rounded">Servis</button>
                      </div>
                    </div>

                    <div className="lg:col-span-2 flex justify-center bg-gray-900 border border-gray-800 p-4 rounded-3xl">
                      <canvas
                        ref={canvasRef}
                        width="300"
                        height="400"
                        className="border border-gray-800 rounded-2xl cursor-crosshair bg-emerald-950 touch-none"
                        onMouseDown={handleCanvasMouseDown}
                        onMouseMove={handleCanvasMouseMove}
                        onMouseUp={handleCanvasMouseUpOrLeave}
                        onMouseLeave={handleCanvasMouseUpOrLeave}
                        onTouchStart={handleCanvasTouchStart}
                        onTouchMove={handleCanvasTouchMove}
                        onTouchEnd={handleCanvasMouseUpOrLeave}
                      />
                    </div>
                  </div>
                )}
              </main>

              {selectedSlip && (
                <div className="fixed inset-0 bg-black/90 backdrop-blur-md z-50 flex items-center justify-center p-4">
                  <div className="relative max-w-md w-full bg-gray-900 border border-gray-800 rounded-3xl p-6">
                    <button onClick={() => setSelectedSlip(null)} className="absolute top-4 right-4 text-white font-bold">✕</button>
                    <img src={selectedSlip} alt="Bukti Kas GKI-BS" className="w-full h-auto rounded-xl" />
                  </div>
                </div>
              )}
            </div>
          );
        }

        // Render application
        const rootElement = document.getElementById('root');
        const root = ReactDOM.createRoot(rootElement);
        root.render(<App />);
    </script>
